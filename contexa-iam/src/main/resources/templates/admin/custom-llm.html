Implementation Code (Client: 10 Lines)

const client = new ContexaStreaming.StreamingClientBuilder()
    .withModalUI({ headerText: 'AI Processing...' })
    .build();

document.getElementById('submit-btn').onclick = async () => {
    const query = document.getElementById('query-input').value;
    const result = await client.streamWithAdapter('/admin/custom-llm/api/stream', { query });
    document.getElementById('result').innerHTML = result.answer || JSON.stringify(result, null, 2);
};
